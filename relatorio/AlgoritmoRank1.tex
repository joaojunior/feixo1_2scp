\begin{algorithm}
\Entrada{Instância para o $SCP$}
\Saida{Cobertura de menor custo encontrada}
\Inicio{ \nonumber
        x $\leftarrow$ Resolução da relaxação linear da instância $SCP$\\
        \While{x não for inteiro ou o tempo máximo ocorra}{
            \If{Existe algum corte de Chvátal-Gomory rank-1 que corta x?}{
                Adicione o corte ao modelo relaxado \\
                x $\leftarrow$ Resolução da relaxação linear da instância $SCP$\\
            }
            \Else{
                break \\
            }
        }
        \Retorna Melhor Solução Encontrada \\
}            
\end{algorithm}
